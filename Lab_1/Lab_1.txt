CREATE TABLE JobSeekers (
  JobSeekerID INT PRIMARY KEY IDENTITY,
  FirstName NVARCHAR(50) NOT NULL,
  LastName NVARCHAR(50) NOT NULL,
  Email NVARCHAR(100) NOT NULL,
  PhoneNumber NVARCHAR(20) NOT NULL,
  DateOfBirth DATE NOT NULL,
  Address NVARCHAR(100) NOT NULL,
  Education NVARCHAR(100) NOT NULL,
  WorkExperience NVARCHAR(100) NOT NULL,
  Skills NVARCHAR(100) NOT NULL
);

CREATE TABLE Employers (
  EmployerID INT PRIMARY KEY IDENTITY,
  CompanyName NVARCHAR(100) NOT NULL,
  ContactPerson NVARCHAR(100) NOT NULL,
  Email NVARCHAR(100) NOT NULL,
  PhoneNumber NVARCHAR(20) NOT NULL,
  Address NVARCHAR(100) NOT NULL
);

CREATE TABLE Vacancies (
  VacancyID INT PRIMARY KEY IDENTITY,
  EmployerID INT FOREIGN KEY REFERENCES Employers(EmployerID),
  PositionID INT FOREIGN KEY REFERENCES Positions(PositionID),
  SalaryRange MONEY NOT NULL,
  Description NVARCHAR(MAX) NOT NULL,
  State BINARY NOT NULL,
);

CREATE TABLE Positions (
  PositionID INT PRIMARY KEY IDENTITY,
  PositionName NVARCHAR(100) NOT NULL
);

CREATE TABLE Applications (
  ApplicationID INT PRIMARY KEY IDENTITY,
  JobSeekerID INT FOREIGN KEY REFERENCES JobSeekers(JobSeekerID),
  VacancyID INT FOREIGN KEY REFERENCES Vacancies(VacancyID),
  ApplicationDate DATE NOT NULL,
  Resume NVARCHAR(MAX) NOT NULL,
  CoverLetter NVARCHAR(MAX) NOT NULL
);
